struct Sinusoid {

	period      f64
	mean_value  f64
	amplitude   f64
	phase_shift f64

	frequency    f64
	angular_freq f64
	time_shift   f64

	b []f64

Sinusoid implements the sinusoid equation: y(t) = a0 + c1⋅cos(omega_0⋅t + theta) [essential-form] y(t) = a0 + a1⋅cos(omega_0⋅t) + b1⋅sin(omega_0⋅t) [basis-form] a1 = c1⋅cos(theta) b1 = -c1⋅sin(theta) theta = arctan(-b1 / a1) if a1<0, theta += π c1 = sqrt(a1² + b1²)
